import java.util.*;

public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int a, b, t, sum;
        t=sc.nextInt();
        for(int i=1;i<=t;i++){
            a=sc.nextInt();
            b=sc.nextInt();
            if(0<a && a<10 && 0<b && b<10){
                sum = a+b;
                System.out.println("Case #"+i+": "+a+" + "+b+" = "+sum);
            }
        }
    }
}

        if(1<=n && n<=100){
            for(int i=1;i<=n;i++){
                for(int j=1;j<=i;j++){

import java.util.*;

public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        if(1<=n && n<=100){
            for(int i=1; i<=n; i++){
               for(int j=1; j<=n-i; j++){
                   System.out.print(" ");
               }
               for(int o=n-i; o<n; o++){
                   System.out.print("*");
               }
               System.out.print("\n");
            }
        }
    }
}

# 10871번


![](https://doozi316.github.io/assets/images/Algorithm7/4.png)

```
import java.util.*;

public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int n, x;
        n = sc.nextInt();
        x = sc.nextInt();
        
        if(1<=n && n<=10000 && 1<=x && x<=10000){
            int[] a = new int[n];
            for(int i=0; i<n; i++){
                a[i] = sc.nextInt();
            }
            for(int j=0; j<n; j++){
                if(a[j]<x){
                    System.out.print(a[j]+" ");
                }
            }
        }
    }
}
```
