import java.util.*;

public class Main {
    public static void main(String args[]) {
        int num[] = new int[10];
        int count[] = new int[42];
        int result=0;
        Scanner sc = new Scanner(System.in);

        for(int i=0;i<10;i++){
            num[i]=sc.nextInt();
        }
        for(int i=0;i<10;i++){
            if(num[i]<=1000 || num[i]>=0){
                count[num[i]%42]++;
            }
        }
       for(int i=0;i<42;i++){
           if(count[i]!=0){
               result+=1;
           }
       }
        System.out.print(result);
    }
}

import java.util.*;

public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int C = sc.nextInt();
        for(int i=0;i<C;i++){
            int sum=0;
            int N, score[];
            N = sc.nextInt();
            score = new int[N];
            for(int j=0;j<N;j++){
                score[j] = sc.nextInt();
                sum += score[j];
            }
            int avg = sum/N;
            float winner = 0;
            for(int j=0;j<N;j++){
                if(avg<score[j]){
                    winner +=1;
                }
            }
            float result = winner/N*100;
            System.out.printf("%.3f", result);
            System.out.println("%");
        }
    }
}
