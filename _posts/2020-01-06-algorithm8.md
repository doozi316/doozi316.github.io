# 10925번

![](https://doozi316.github.io/assets/images/Algorithm8/1.png)
```
import java.util.*;

public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int a, b;
        while(true){
            a = sc.nextInt();
            b = sc.nextInt();
            if(a==0 && b==0) break;
            if(0<a && a<10 & 0<b && b<10){
                System.out.println(a+b);
            }
        }
    }
}
```

# 1110번


![](https://doozi316.github.io/assets/images/Algorithm8/3.png)

```
import java.util.*;

public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int sum = 0;
        int i, j, o, x;
        int a = sc.nextInt();
        if(0<=a && a<=99){
            int n = a;
            while(true){
            i = n/10;
            j = n%10;
            o = (i+j)%10;
            x = j*10;
            n = o+x;
            sum += 1;
            if(a==n) break;
            }
            System.out.print(sum);
        }
    }
}
```
