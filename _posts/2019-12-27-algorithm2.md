# 1001번

![](https://doozi316.github.io/assets/images/Algorithm2/2.png)

```
import java.util.*;

public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        
        int A, B;
        A = sc.nextInt();
        B = sc.nextInt();
        if( 0 < A && B < 10) {
            System.out.print(A - B);
        } else {
            System.out.print("0<A, B<10이어야합니다.");
        }
    }
}
```
